library(GIFT)

#This example was generated from a region containing four genes in chr5. 
#The number of cis-SNPs is 63, 23, 41 and 96 for RASA1, COX7C, CCNH and TMEM161B, respectively
#We selected RASA1 as the causal gene in the region with the effect size being sqrt(0.1)

#load the simulated scaled genenotype matrix in eQTL data (e.g. from GEUVADIS data)
Zx<-read.table("Zx.txt")
Zx<-as.matrix(Zx)

#load the simulated scaled genenotype matrix in GWAS data (e.g. from UK data)
Zy<-read.table("Zy.txt")
Zy<-as.matrix(Zy)

#load the simulated exposure or gene expression vector
X<-read.table("X.txt")
X<-as.matrix(X)

#load the simulated phenotype vector
Y<-read.table("Y.txt")$V1

#load the gene names and number of cis-SNPs for each gene
gene=c("RASA1", "COX7C", "CCNH", "TMEM161B")
pindex=c(63, 23, 41, 96)

X<-scale(X)
Y<-scale(Y)
Zx<-scale(Zx)
Zy<-scale(Zy)

#run GIFT
result<-GIFT_individual(X, Y, Zx, Zy, gene, pindex, max_iterin =1000,epsin=1e-4,Cores=1)
